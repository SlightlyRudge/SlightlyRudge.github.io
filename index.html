async function loadCategories() { try { const response = await fetch('https://db.slightlyrudge7.workers.dev/api/categories'); if (!response.ok) { throw new Error(`HTTP error! Status: ${response.status}`); } const data = await response.json(); console.log("API 数据：", data); // 控制台打印数据，调试用 const container = document.getElementById('categories'); container.innerHTML = ''; // 清空内容，防止重复渲染 for (const category in data) { const section = document.createElement('div'); section.className = 'category-container'; const title = document.createElement('div'); title.className = 'category'; title.innerText = category; section.appendChild(title); const buttonContainer = document.createElement('div'); buttonContainer.className = 'buttons'; section.appendChild(buttonContainer); data[category].forEach(name => { const btn = document.createElement('a'); btn.className = 'btn'; btn.innerText = name; btn.href = `/list.html?category=${encodeURIComponent(category)}&tag=${encodeURIComponent(name)}`; buttonContainer.appendChild(btn); }); container.appendChild(section); } } catch (error) { console.error("加载分类失败:", error); document.getElementById('categories').innerHTML = '

无法加载分类数据

'; } } // 调用函数，加载分类 loadCategories();
